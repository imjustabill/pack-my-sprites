  code/application.cpp
  code/gimp_interface.cpp
  code/image_generator.cpp
  code/main.cpp
  code/makefile_generator.cpp
  code/parser.cpp
  code/plist_generator.cpp

  code/spritedesc.cpp
  code/spritedesc_collection.cpp
  code/spritepos_generator.cpp
  code/xcf_info.cpp

  code/working_directory.cpp

  node_parser/code/node_parser.cpp
  node_parser/code/node_parser_file.cpp

#-------------------------------------------------------------------------------
# Boost
include(FindBoost)

find_package(
  Boost 1.42 REQUIRED COMPONENTS filesystem system
  )

#-------------------------------------------------------------------------------
# Claw
find_package(libclaw 1.7.4)

#-------------------------------------------------------------------------------
add_executable(
  ${SDC_EXECUTABLE_NAME}
  ${SDC_SOURCE_FILES}
  )

if( SDC_DEFAULT_SCHEME_PATH )
  set(
    SDC_COMPILE_DEFINITIONS
    ${SDC_COMPILE_DEFINITIONS}
    "SDC_DEFAULT_SCHEME_PATH=${SDC_DEFAULT_SCHEME_PATH}"
    )
endif()

set_target_properties(
  ${SDC_EXECUTABLE_NAME}
  PROPERTIES
  COMPILE_DEFINITIONS
  "${SDC_COMPILE_DEFINITIONS}"
  )

target_link_libraries(
  ${SDC_EXECUTABLE_NAME}
  ${CLAW_APPLICATION_LIBRARIES}
  ${Boost_FILESYSTEM_LIBRARY}
  ${Boost_SYSTEM_LIBRARY}
  )

install(
  TARGETS ${SDC_EXECUTABLE_NAME}
  DESTINATION ${SDC_INSTALL_EXECUTABLE_DIR}
  PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE
  GROUP_READ GROUP_EXECUTE
  WORLD_READ WORLD_EXECUTE
  )
